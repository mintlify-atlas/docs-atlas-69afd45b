  <h2 className="text-2xl sm:text-3xl font-bold text-gray-900 dark:text-white mb-4">
    Quick Start
  </h2>
  <p className="text-base text-gray-600 dark:text-gray-400 mb-6">
    Get your first lookup running in under a minute.
  </p>

  <Steps>
    <Step title="Install the package">
      Choose your language and install via your package manager.
      
      <CodeGroup>
      ```bash npm
      npm install @aialok/lakhua
      ```
      ```bash bun
      bun add @aialok/lakhua
      ```
      ```bash pip
      pip install lakhua
      ```
      ```bash go
      go get github.com/aialok/lakhua/libs/go
      ```
      </CodeGroup>
    </Step>

    <Step title="Import and use">
      Call the geocode function with latitude and longitude.
      
      <Tabs>
        <Tab title="JavaScript">
          ```javascript
          import { geocode } from "@aialok/lakhua";

          const result = geocode(12.9716, 77.5946);
          console.log(result.city, result.state);
          // Bengaluru Karnataka
          ```
        </Tab>
        <Tab title="Python">
          ```python
          from lakhua import geocode

          result = geocode(12.9716, 77.5946)
          print(result.city, result.state)
          # Bengaluru Karnataka
          ```
        </Tab>
        <Tab title="Go">
          ```go
          import "github.com/aialok/lakhua/libs/go"

          result := lakhua.Geocode(12.9716, 77.5946, nil)
          fmt.Println(result.City, result.State)
          // Bengaluru Karnataka
          ```
        </Tab>
      </Tabs>
    </Step>

    <Step title="Explore options">
      Configure resolution, fallback behavior, and debug logging.
      
      <Tabs>
        <Tab title="JavaScript">
          ```javascript
          const result = geocode(28.6139, 77.2090, {
            resolution: 5,
            fallback: true,
            debug: false
          });
          ```
        </Tab>
        <Tab title="Python">
          ```python
          from lakhua import geocode, GeocodeOptions

          result = geocode(28.6139, 77.2090, GeocodeOptions(
              resolution=5,
              fallback=True,
              debug=False
          ))
          ```
        </Tab>
        <Tab title="Go">
          ```go
          result := lakhua.Geocode(28.6139, 77.2090, &lakhua.GeocodeOptions{
              Resolution: 5,
              Fallback:   lakhua.BoolPtr(true),
              Debug:      false,
          })
          ```
        </Tab>
      </Tabs>
    </Step>
  </Steps>
</div>
